"""This file serves the emotion detection endpoint."""

from flask import Flask, request, jsonify
from EmotionDetection import emotion_detector

app = Flask(__name__)

@app.route('/emotionDetector', methods=['GET'])
def get_emotion():
    """This file returns the detected emotion."""
    sentence = request.args.get('sentence')

    res = emotion_detector(sentence)
    print("Ankit res")
    print(res)
    if res and res['dominant_emotion'] is None:
        return "Invalid text! Please try again!."

    return jsonify(res)
